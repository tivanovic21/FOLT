KREIRAJ I POKRENI TVRTKA SLIKU
docker build -f Dockerfile.faza_3.1 -t posluzitelj_tvrtka:faza_3 .
docker run -it -d -p 8000:8000 --name=PosluziteljTvrtka posluzitelj_tvrtka:faza_3

POKRENI JMETER
docker run -it -d --mount type=bind,source="$(pwd)"/podaci,target=/usr/app/podaci --name=jmeter jmeter:faza_3

ZAUSTAVI KONTEJNERE I OBRIŠI
docker stop PosluziteljTvrtka
docker rm PosluziteljTvrtka
docker rm jmeter

POKRENI KONTEJNER
docker start PosluziteljTvrtka

PREGLED AKTIVNIH DOCKER CONTAINERA
docker ps

UĐI U DOCKER CONTAINER
docker exec -it PosluziteljTvrtka /bin/bash

PREGLED AKTIVNIH PROCESA KAD SI UNUTAR DOCKER CONTAINERA
ps -ef

PREGLED SLIKA
docker image ls
